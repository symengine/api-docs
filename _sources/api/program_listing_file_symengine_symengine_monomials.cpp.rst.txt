   // This is the fastest implementation:
   void monomial_mul(const vec_int &A, const vec_int &B, vec_int &C)
   {
       size_t n = A.size();
       for (size_t i = 0; i < n; ++i) {
           C[i] = A[i] + B[i];
       }
   }
